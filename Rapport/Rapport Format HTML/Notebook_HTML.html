class KNN:
    
    def __init__(self,k=1):
        self.k=k
        
    def Train(self,X_train,Y_train):
        self.X_train=X_train
        self.Y_train=Y_train
        
    def Distance(self,ptTest,ptTrain):
        #On commence par dfinir une fonction de calcul de distance. Ici on choisit la distance euclidienne
        distance=0
        for k in range(len(ptTrain)):
            distance+=(ptTrain[k]-ptTest[k])**2
        return np.sqrt(distance)
    
    
    def GetLabel(self,instance):
        #On definit la fonction GetLabel permettant de prdir l'tiquette d'une instance
        #On rcupre la distance de l'instance  chaque point
        distances=[self.Distance(instance,x) for x in self.X_train ]
        #On rcupre les index de la liste trie
        indexes=np.argsort(distances)[:self.k]
        #On rcupre les label tris selon ces indexs
        Top_Neighbours=np.array([self.Y_train[i] for i in indexes])

        #On renvoie le plus prsent
        return max(labels_Class,key=lambda x:np.sum(Top_Neighbours==x))

    def GetLabels(self,X):
        #On rcupre ici le label de chaque instance
        labels=[self.GetLabel(x) for x in X]
        return np.array(labels)
    
    def Precision(self,X_test,Y_test,MC):
        #On compare nos label estims aux rel labels et on en dduit une precision
        labels=self.GetLabels(X_test)
        precision=np.sum(labels==Y_test)/len(labels)*100
        
        #Mode Affichage
        if MC:
            fig = plt.figure(figsize=(5,5))
            ax = fig.add_subplot(111)
            im = ax.imshow(confusion_matrix(Y_test,labels))
            fig.colorbar(im)
            plt.title("Matrice de confusion")
            plt.xlabel("Predicted")
            plt.ylabel("Actual")
        
        return precision

Optimisation du model¶

Maintenant que notre modle est construit et qu'il nous fournit des resultats acceptables, nous pouvons essayer de rgler le paramtre k afin d'augmenter notre score. Pour cela on va sparer nos donnes non pas en deux mais en 3 sets. Un pour l'entrainement, un pour la validation (optimisation de k ) et un dernier pour le test final.

Cependant, plusieurs dcoupages de nos donnes sont possibles pour choisir notre set de validation. On va donc utiliser la mthode de cross validation qui permet d'explorer tout les dcoupages possibles pour notre set et de retourner chaque fois son score.
Ici on va utilise le dcoupage des KFolds

In [8]:
def CrossValidation(k,X_train,Y_train,nFolds=5):
    scoresValidation=[]
    
    model=KNN(k)
    #On dcoupe en n folds notre data set
    X_Splits=np.array_split(X_train,nFolds)
    Y_Splits=np.array_split(Y_train,nFolds)
    
    #Pour chaque fold on va mettre  jour le set d'entrainement et de validation et on va calculer le score
    for i in range(nFolds):
        print("\nCalcul du fold n(",i+1,"/",nFolds,")..")
        X_validation=X_Splits[i]
        Y_validation=Y_Splits[i]
        X_newTrain=np.concatenate(np.delete(X_Splits,i,0))
        Y_newTrain=np.concatenate(np.delete(Y_Splits,i,0))
        
        model.Train(X_newTrain,Y_newTrain)
        scoresValidation.append(model.Precision(X_validation,Y_validation,False))
        
    #On renvoi le tableau des scores
    return np.array(scoresValidation)

def Optimise(listeK,X_train,Y_train):

    #On applique pour chaque k la mthode de validation croise
    for k in listeK:
        print("===========================Test k=",k,"==================================")
        
        #On prend ici la moyenne des scores de chaque fold
        accuracy=CrossValidation(k,X_train,Y_train,5).mean()

        print("\n Accuracy =",accuracy)
    return scoresValidation
